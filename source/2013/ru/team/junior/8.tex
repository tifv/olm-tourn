\problem\problemscore{10}
Положительные числа $x$, $y$ и $z$ таковы, что $x + y + z = 1$.
Докажите неравенство
\[
    \frac{x y}{\sqrt{z + x y}}
    +
    \frac{y z}{\sqrt{x + y z}}
    +
    \frac{z x}{\sqrt{y + z x}}
\leq
    \frac{1}{2}
.\]

\solution Решение \texttt{team/junior/8.tex}.
\endproblem
