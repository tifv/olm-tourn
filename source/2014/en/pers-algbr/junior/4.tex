\problem
For natural $n$, $k$ prove the inequality:
\[
    1 + \frac{1}{2} + \ldots + \frac{1}{n k}
<
    \left(
        1 + \frac{1}{2} + \ldots + \frac{1}{n}
    \right)
    +
    \left(
        1 + \frac{1}{2} + \dots + \frac{1}{k}
    \right)
    -
    \frac{1}{2}
\;.\]
\solution
\endproblem
