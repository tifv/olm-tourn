\problem
Let
\begin{gather*}
    x_1 + x_2 + x_3 + x_4 = y_1 + y_2 + y_3 + y_4
,\\
    x_1^2 + x_2^2 + x_3^2 + x_4^2
=
    y_1^2 + y_2^2 + y_3^2 + y_4^2
,\\
    x_1^3 + x_2^3 + x_3^3 + x_4^3
=
    y_1^3 + y_2^3 + y_3^3 + y_4^3
.\end{gather*}
Prove that
\[
    (x_1 - y_2) (x_1 - y_3) (x_1 - y_4)
=
    (y_1 - x_2) (y_1 - x_3) (y_1 - x_4)
.\]

\solution

\endproblem
