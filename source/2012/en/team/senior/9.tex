\problem\problemscore{8}
Positive numbers $a$, $b$, $c$ and $d$ are such that $a + b + c + d = 1$.
Prove that
\[
   \frac{a^3}{(1 + b) (1 - c)}
   +
   \frac{b^3}{(1 + c) (1 - d)}
   +
   \frac{c^3}{(1 + d) (1 - a)}
   +
   \frac{d^3}{(1 + a) (1 - b)}
\geq
   \frac{1}{15}\:.
\]
\solution
\endproblem
